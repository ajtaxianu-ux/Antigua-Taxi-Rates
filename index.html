    <script>

        // The core logic to calculate and display the fare

        const hotelSelect = document.getElementById('hotel-select');

        const peopleSelect = document.getElementById('people-select');

        const totalFareDisplay = document.getElementById('total-fare');


        function calculateFare() {

            const selectedOption = hotelSelect.options[hotelSelect.selectedIndex];

            const baseFare = selectedOption.dataset.price ? parseFloat(selectedOption.dataset.price) : 0;

            const multiplier = parseFloat(peopleSelect.value);

            const totalFare = baseFare * multiplier;

            totalFareDisplay.textContent = `$${totalFare}`;

        }


        hotelSelect.addEventListener('change', calculateFare);

        peopleSelect.addEventListener('change', calculateFare);


        // Initial calculation on page load

        calculateFare();


        // Booking form submission logic

        document.getElementById('booking-form').addEventListener('submit', function(event) {

            event.preventDefault();

            

            const name = document.getElementById('name').value;

            const email = document.getElementById('email').value;

            const hotel = hotelSelect.options[hotelSelect.selectedIndex].text;

            const people = peopleSelect.options[peopleSelect.selectedIndex].text;

            const totalFare = totalFareDisplay.textContent;


            const subject = encodeURIComponent(`Booking Inquiry from Website: ${name}`);

            const body = encodeURIComponent(

                `Hello Aj's Taxi Antigua Adventures,\n\n` +

                `I would like to book a taxi for my airport transfer.\n\n` +

                `Name: ${name}\n` +

                `Email: ${email}\n` +

                `Hotel: ${hotel}\n` +

                `Number of People: ${people}\n` +

                `Estimated Total Fare: ${totalFare}\n\n` +

                `Thank you.`

            );


            // This line needs to be updated with your actual email address

            window.location.href = `mailto:[Your Email Address]?subject=${subject}&body=${body}`;


            const statusMessage = document.getElementById('status-message');

            statusMessage.textContent = 'Thank you! Your email client should now be open to send the inquiry.';

            statusMessage.classList.remove('hidden');


            setTimeout(() => {

                statusMessage.classList.add('hidden');

            }, 10000);

        });
